## Store setup

```typescript
// src/app/store.ts
import { configureStore } from '@reduxjs/toolkit'
import { ordersApi } from '../features/orders/orders.api'
import ordersReducer from '../features/orders/orders.slice'
import uiReducer from '../features/ui/ui.slice'

export const store = configureStore({
  reducer: {
    // RTK Query API reducers
    [ordersApi.reducerPath]: ordersApi.reducer,

    // Feature slices (client-side state)
    orders: ordersReducer,
    ui: uiReducer,
  },
  middleware: (getDefaultMiddleware) =>
    getDefaultMiddleware().concat(
      ordersApi.middleware,  // required for RTK Query caching
    ),
})

// Typed helpers — always use these instead of raw hooks
export type RootState   = ReturnType<typeof store.getState>
export type AppDispatch = typeof store.dispatch
```

```typescript
// src/app/hooks.ts — typed versions of useDispatch and useSelector
import { useDispatch, useSelector, TypedUseSelectorHook } from 'react-redux'
import type { RootState, AppDispatch } from './store'

export const useAppDispatch: () => AppDispatch = useDispatch
export const useAppSelector: TypedUseSelectorHook<RootState> = useSelector
```

## Slice — createSlice

```typescript
// src/features/orders/orders.slice.ts
import { createSlice, PayloadAction } from '@reduxjs/toolkit'

interface OrdersState {
  selectedOrderId: string | null
  filters: {
    status: 'all' | 'pending' | 'shipped' | 'delivered'
    dateRange: { from: string; to: string } | null
  }
}

const initialState: OrdersState = {
  selectedOrderId: null,
  filters: {
    status: 'all',
    dateRange: null,
  },
}

const ordersSlice = createSlice({
  name: 'orders',
  initialState,
  reducers: {
    selectOrder(state, action: PayloadAction<string | null>) {
      state.selectedOrderId = action.payload   // Immer allows mutation syntax
    },
    setStatusFilter(state, action: PayloadAction<OrdersState['filters']['status']>) {
      state.filters.status = action.payload
    },
    clearFilters(state) {
      state.filters = initialState.filters
    },
  },
})

// Export actions
export const { selectOrder, setStatusFilter, clearFilters } = ordersSlice.actions

// Export selectors
export const selectSelectedOrderId = (state: RootState) => state.orders.selectedOrderId
export const selectOrderFilters    = (state: RootState) => state.orders.filters

export default ordersSlice.reducer
```

## RTK Query — data fetching

RTK Query replaces `useEffect` + `fetch` for all server state:

```typescript
// src/features/orders/orders.api.ts
import { createApi, fetchBaseQuery } from '@reduxjs/toolkit/query/react'

export interface Order {
  id: string
  status: 'pending' | 'shipped' | 'delivered'
  total: number
  createdAt: string
}

export const ordersApi = createApi({
  reducerPath: 'ordersApi',
  baseQuery: fetchBaseQuery({
    baseUrl: '/api/v1',
    prepareHeaders: (headers) => {
      const token = localStorage.getItem('auth_token')
      if (token) headers.set('Authorization', `Bearer ${token}`)
      return headers
    },
  }),
  tagTypes: ['Order'],    // for cache invalidation
  endpoints: (builder) => ({
    // Queries (GET)
    listOrders: builder.query<{ data: Order[]; nextCursor: string | null }, { cursor?: string; limit?: number }>({
      query: ({ cursor, limit = 20 } = {}) => ({
        url: '/orders',
        params: { cursor, limit },
      }),
      providesTags: (result) =>
        result
          ? [...result.data.map(({ id }) => ({ type: 'Order' as const, id })), 'Order']
          : ['Order'],
    }),

    getOrder: builder.query<{ data: Order }, string>({
      query: (id) => `/orders/${id}`,
      providesTags: (result, error, id) => [{ type: 'Order', id }],
    }),

    // Mutations (POST/PATCH/DELETE)
    createOrder: builder.mutation<{ data: Order }, Partial<Order>>({
      query: (body) => ({
        url: '/orders',
        method: 'POST',
        body,
      }),
      invalidatesTags: ['Order'],   // invalidate list after create
    }),

    updateOrder: builder.mutation<{ data: Order }, { id: string; body: Partial<Order> }>({
      query: ({ id, body }) => ({
        url: `/orders/${id}`,
        method: 'PATCH',
        body,
      }),
      invalidatesTags: (result, error, { id }) => [{ type: 'Order', id }],
    }),
  }),
})

// Export hooks — auto-generated by RTK Query
export const {
  useListOrdersQuery,
  useGetOrderQuery,
  useCreateOrderMutation,
  useUpdateOrderMutation,
} = ordersApi
```

## Using RTK Query in components

```typescript
// src/features/orders/OrderList.tsx
import { useListOrdersQuery } from './orders.api'
import { useAppSelector, useAppDispatch } from '../../app/hooks'
import { selectOrderFilters, selectOrder } from './orders.slice'

export function OrderList() {
  const filters = useAppSelector(selectOrderFilters)
  const dispatch = useAppDispatch()

  const { data, isLoading, isError, isFetching } = useListOrdersQuery({
    limit: 20,
  })

  if (isLoading) return <Skeleton />
  if (isError)   return <ErrorState />
  if (!data?.data.length) return <EmptyState />

  return (
    <ul>
      {data.data.map((order) => (
        <li key={order.id} onClick={() => dispatch(selectOrder(order.id))}>
          {order.id} — {order.status}
        </li>
      ))}
    </ul>
  )
}
```

```typescript
// Mutation in a form component
import { useCreateOrderMutation } from './orders.api'

export function CreateOrderForm() {
  const [createOrder, { isLoading, error }] = useCreateOrderMutation()

  async function handleSubmit(data: FormData) {
    try {
      const result = await createOrder(data).unwrap()  // unwrap throws on error
      navigate(`/orders/${result.data.id}`)
    } catch (err) {
      // err is the API error response
    }
  }
}
```

## Memoised selectors — createSelector

```typescript
import { createSelector } from '@reduxjs/toolkit'
import { selectOrderFilters } from './orders.slice'
import { ordersApi } from './orders.api'

// Derived state — never store computed values in Redux state
export const selectFilteredOrders = createSelector(
  [
    ordersApi.endpoints.listOrders.select({}),
    selectOrderFilters,
  ],
  (ordersResult, filters) => {
    const orders = ordersResult.data?.data ?? []
    if (filters.status === 'all') return orders
    return orders.filter((o) => o.status === filters.status)
  }
)
```

## createAsyncThunk (for non-API async ops)

Use for side effects that aren't data fetching:

```typescript
// Example: download a file
import { createAsyncThunk } from '@reduxjs/toolkit'

export const downloadOrderCsv = createAsyncThunk(
  'orders/downloadCsv',
  async (orderId: string, { rejectWithValue }) => {
    try {
      const blob = await generateCsvBlob(orderId)
      downloadFile(blob, `order-${orderId}.csv`)
    } catch (err) {
      return rejectWithValue(err.message)
    }
  }
)

// Handle in slice
const ordersSlice = createSlice({
  name: 'orders',
  initialState,
  reducers: { ... },
  extraReducers: (builder) => {
    builder
      .addCase(downloadOrderCsv.pending, (state) => {
        state.downloadStatus = 'loading'
      })
      .addCase(downloadOrderCsv.fulfilled, (state) => {
        state.downloadStatus = 'idle'
      })
      .addCase(downloadOrderCsv.rejected, (state, action) => {
        state.downloadStatus = 'error'
        state.downloadError = action.payload as string
      })
  },
})
```

## Common mistakes

| Mistake | Fix |
|---|---|
| `useDispatch` and `useSelector` directly | Use `useAppDispatch` and `useAppSelector` — they're typed |
| Server state in Redux slice | Use RTK Query — it handles caching, loading, and invalidation |
| Derived data stored in state | Use `createSelector` — compute from existing state |
| Mutating state outside a reducer | Only mutations inside `createSlice` reducers (Immer) are safe |
| One giant slice for everything | One slice per feature/domain |
| Not calling `.unwrap()` on mutations | `.unwrap()` converts RTK's error envelope into a thrown error |
